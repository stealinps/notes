#Big O notation
-Measures the efficiency of an algorithm when the input grows commonly annotate by "n"
-O(1),O(log n),O(n),O(nlogn),O(n²),O(2^n),O(n!)

#Time complexity
-Measured in unit of steps/How many steps it took to complete the algorithm
-O(1) means instantly
-O(n) means it grows linearly 
-O(log n) means it grows slower than O(n) [aka logarithmically]
-O(n²) means it grows quadratically
-O(nlogn) means it grows faster than O(log n) but slower than O(n²)
-O(2^n) means it grows exponentially faster than O(n²)
-O(n!) grows faster than O(2^n) at big n value

#include <iostream>
int main(){
	std::cout<<"Hello world!"<<std::endl;
	int n;
	std::cout<<"Enter 0 for asc/1 for des"<<std::endl;
	std::cin>>n; 
	switch (n)
	{
		case 0:
			for(int i=0;i<=50;i++){
				std::cout<<i<<std::endl;
			}
			break;
		case 1:
			for(int i=50;i>=0;i--){
				std::cout<<i<<std::endl;
			}
			break;
		case 2:
			for(int i=1;i<=50;i=i+2){
				std::cout<<i<<std::endl;
			}
			break;
		case 3:
			for(int i=0;i<=50;i=i+2){
				std::cout<<i<<std::endl;
			}
			break;
		default:
			std::cout<<"Invalid input ";
			break;
}
}
